from django.shortcuts import render
from .models import User, UserId

def index(request):
    """hp: xxx"""
    return render(request, 'app_xxx/index.html')

def topics(request):
    """hp: xxx"""
    users = User.objects.order_by('date_added')
    context = {'users': users}
    return render(request, 'app_xxx/users.html', context)

def user(request, topic_id):
    """hp: xxx"""
    user = User.objects.get(id=user_id)     # in *.html: user.id
    entries = user.entry_set.order_by('-date_added')
    context = {'user': user, 'entries': entries}
    return render(request, 'app_xxx/user.html', context)
